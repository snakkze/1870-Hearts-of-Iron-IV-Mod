		}

			original_tag = PRS
			has_completed_focus = PRS_habsburg_enemy
		}

		icon = mission_icon
		
		complete_effect = {
			every_country = { 
				limit = { 
					OR = {
						is_german_minor = yes

						original_tag = HES
					}
				}
				PRS = {
					if = {
						limit = {
							NOT = {
								has_war_with = PREV
							}
						}
						declare_war_on = { type = annex_everything target = PREV }
					}
				}
			}
			BAV = {
				create_faction = fpw_GERMAN_LEAGUE
			}
			every_country = { 
				limit = { 
					AND = {
						OR = {
							is_german_minor = yes

						}
					}
