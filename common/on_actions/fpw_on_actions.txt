on_actions = {
	on_generate_wargoal = {
		effect = {
			if = {
				limit = {
					FROM = {
						original_tag = TRB
					}
				}
				ROOT = {
					remove_wargoal = {
						type = all
						target = TRB
					}
				}
			}
		}
	}
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					FROM = {
						original_tag = TRB
					}
				}
				ROOT = {
					white_peace = TRB
				}
			}
		}
	}
	on_war = {
		effect = {
			if = {
				limit = {
					THIS = { 
						original_tag = TRB
					}
				}
				every_country = {
					limit = {
						has_war_with = TRB
					}
					white_peace = TRB
				}
			}
		}
	}
	on_declare_war  = {
		effect = {
			if = {
				limit = {
					FROM = { 
						original_tag = TRB
					}
				}
				ROOT = {
					white_peace = TRB
				}
			}
		}
	}

	on_startup = {
		effect = {
			every_country = {
				limit = {
					NOT = {
						tag = TRB

				}
				add_opinion_modifier = {
    					target = TRB
    					modifier = fpw_tribal_state_modifier
				}
				reverse_add_opinion_modifier = {
					target = TRB
					modifier = fpw_tribal_state_modifier
				}
			}
			clear_array = global.prestige_list_array
			every_country = {
				if = {
					limit = {
						NOT = { is_in_array = { global.prestige_list_array = THIS } }
					}
					add_to_array = { global.prestige_list_array = THIS }
				}
			}
			calculate_prestige = yes
			TRB = {
				set_variable = { THIS.country_prestige = 0 }
			}
			update_position = yes

			every_country = {
				set_variable = { THIS.extra_colonizing_speed = 1 }
			}

			every_country = {
				if = {
					limit = {
						is_debug = yes
					}
					set_variable = { THIS.colonists = 100 }
				} else = { set_variable = { THIS.colonists = 1 } }
			}

			if = {
				limit = {
					has_start_date < 1866.1.2
					has_start_date > 1865.12.31
				}
				set_global_flag = SUEZ_CANAL_NOT_BUILT
				# German Confederation Stuff
				every_state = {
					limit = {
						OR = {
							is_claimed_by = PRS
							is_core_of = FRK
							is_core_of = BRU
							is_core_of = WRT

							is_core_of = SXW

							is_core_of = SCG
							is_core_of = SBL
							is_core_of = OLD
							is_core_of = SBG
							is_core_of = OLD
							is_core_of = NAS
							is_core_of = MKS
							is_core_of = MCK
							is_core_of = LUE
							is_core_of = LIP
							is_core_of = HOH
							is_core_of = HNV
							is_core_of = HEK
							is_core_of = HES
							is_core_of = HAM
							is_core_of = BRE
							is_core_of = BAV
							is_core_of = BAD
							is_core_of = ANH
							is_core_of = LUX
							AND = {
								is_core_of = PRS
								NOT = {
									state = 86
									state = 807
									state = 85	
									state = 5
									state = 763
									state = 188
									state = 909
									state = 912
								}
							}
							AND = {
								is_core_of = AUS
								NOT = {	
										is_core_of = CRO
										is_core_of = SER
										is_core_of = SLO
										is_core_of = HUN
										is_core_of = ROM
										is_core_of = POL
										is_core_of = SLO
										state = 159
										state = 160
										state = 664
										state = 73
									
								}
							}
						}
					}
					set_state_flag = german_confederation_state
				}
				every_state = {
					limit = {
						is_owned_by = TRB
					}
					set_state_flag = tribe_state
				}
			}
		}
		effect = { #for secondary characters, MUST ADD here for it to appear on start
			if = {
				limit = {
				has_start_date < 1866.1.2
				}

				set_country_flag = fpw_secondary_leader_jap_komei
				}
			}
			PRS = {
				set_country_flag = fpw_secondary_leader_prs_wilhelm_I
			}
			FIN = {

			}

			}

				set_country_flag = fpw_secondary_leader_eng_victoria
			}	
			KOR = {
				set_country_flag = fpw_secondary_leader_kor_gojong_young
			}
			TUR = {
				set_country_flag = fpw_secondary_leader_tur_abdulaziz
			}	
			CHI = {
				set_country_flag = fpw_secondary_leader_chi_tongzhi
			}				
		}
	}
	
	on_monthly = {
		effect = {
			clear_array = global.prestige_list_array
			every_country = {
				if = {
					limit = {
						NOT = { is_in_array = { global.prestige_list_array = THIS } }
					}
					add_to_array = { global.prestige_list_array = THIS }
				}
			}
			calculate_prestige = yes
			TRB = {
				set_variable = { THIS.country_prestige = 0 }
			}
			update_position = yes
		}
	}

	on_daily = {
		effect = {
			# Great War / Crisis
			if = {
				limit = {
					AND = {
						NOT = {
							has_global_flag = crisis_trigger
						}
						any_country = {
							AND = {
								is_in_array = {
									array = global.top_seven_prestige_array
									value = THIS
								}
								original_tag = PRS
								#check_variable = { has_added_tension_amount > 49 }
								#threat > 0.5
							}
						}
					}
				}	
				clear_array = global.gw_initiator_array
				clear_array = global.gw_target_array
				clear_array = global.gw_initiator_supporters_array
				clear_array = global.gw_target_supporters_array

				every_country = {
					limit = {
						is_in_array = {
							array = global.top_seven_prestige_array
							value = THIS
						}
						NOT = {
							# threat > 0.5, thus being the targetted country
							# not threat, it is has_added_tension_amount variable
							# in faction with target
							has_country_flag = fpw_crisis_triggered
							original_tag = PRS
						}
					}
					if = {
						limit = {
							is_ai = no
						}
						set_country_flag = fpw_crisis_triggered
						# scripted loc for multiplayer, if there is a initiator already the loc will say to join them, player gets immediet event stating what happened cause not bothered complicating this
						country_event = fpw_crisis.1
					}
					else = {
						set_country_flag = fpw_crisis_triggered
						# newspaper flavor event for the ai, i know sounds weird but to inform the player and the ai the newspaper will say something like "xy has started an international crisis against xy and parliaments all over the world are currently
						#																																    what side the support"
						country_event = { id = fpw_crisis.1 days = 14 random_days = 6 } 
					}
				}
				set_global_flag = crisis_trigger
			}
		}
	}

	on_startup = {
		effect = {
			every_country = {
				country_event = {
					id = generic.26
					days = 1
				}
			}
		}
	}

	
	## Dynamic State Names
	
	on_state_control_changed = {
		effect = {
			IF = {
				limit = {
					FROM.FROM = { 
						state = 28 
					}
					ROOT = { original_tag = PRS }
					FROM = { 
						NOT = {
							original_tag = PRS
						}	
					}
				}
				28 = { 
					set_state_name = "Elsass–Lothringen" 
				}
				set_province_name = {
					id = 9503
					name = "Straßburg" 
				}
				set_province_name = {
					id = 678
					name = "Mülhausen" 
				}
			}
			IF = {
				limit = {
					FROM.FROM = { 
						state = 28
					}
					NOT = {
						original_tag = PRS
					}
				}
				28 = { 
					reset_state_name = yes
				}
				reset_province_name = 9503
				reset_province_name = 678
			}
		}
	}
	
	on_state_control_changed = { #RUBBER SAPLINGS
		effect = {
			IF = {
				limit = {
					FROM.FROM = { 
						has_resources_amount = {
						resource = rubber
						amount > 0
						}		
					}
					ROOT = {
						NOT = {
						has_country_flag = fpw_access_to_rubber_tree_saplings
						}
					}
				}
				ROOT = {
				set_country_flag = fpw_access_to_rubber_tree_saplings
				}
			}
		}
	}
